  const ext = filePath.toLowerCase();

private get isMarkdownFile(): boolean {
  return this.path.toLowerCase().endsWith('.md') || 
         this.path.toLowerCase().endsWith('.markdown') ||
         this.path.toLowerCase().includes('readme');
}

private getLanguageExtension(path: string): Extension {
  const ext = path.split('.').pop()?.toLowerCase() || '';
  
  switch (ext) {
    case 'js':
    case 'jsx':
      return javascript({ jsx: true });
    case 'ts':
    case 'tsx':
      return javascript({ jsx: true, typescript: true });
    case 'md':
    case 'markdown':
      return markdown({
        base: markdownLanguage,
        codeLanguages: languages,
        addKeymap: true
      });
    case 'html':
    case 'htm':
      return html();
    case 'css':
    case 'scss':
    case 'less':
      return css();
    case 'json':
      return json();
    case 'py':
      return python();
    case 'xml':
      return xml();
    case 'yaml':
    case 'yml':
      return yaml();
    case 'sql':
      return sql();
    // Use legacy modes for other languages
    default:
      // Check if we have legacy mode for this extension
      try {
        const legacyMode = this.getLegacyMode(path);
        if (legacyMode) {
          return legacyMode;
        }
      } catch (e) {
        // Fall through to plain text
      }
      return [];
  }
}

private getLegacyMode(path: string): Extension | null {
  // Implementation for legacy modes (already exists in current codebase)
  // This uses @codemirror/legacy-modes which is already installed
  const ext = path.split('.').pop()?.toLowerCase() || '';
  
  // Import legacy modes (these imports already exist)
  import { markdown as legacyMarkdown } from "@codemirror/legacy-modes/mode/markdown";
  import { javascript as legacyJs } from "@codemirror/legacy-modes/mode/javascript";
  import { python as legacyPython } from "@codemirror/legacy-modes/mode/python";
  // ... other legacy mode imports
  
  // Return appropriate legacy mode
  return null; // Simplified for this plan
}

private createMarkdownPreviewExtension(content: string): Extension {
  // Simple inline preview for headings and code blocks
  const decorations: any[] = [];
  const lines = content.split('\n');
  let pos = 0;
  
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];
    const lineStart = pos;
    const lineEnd = lineStart + line.length;
    
    // Preview headings (h1-h6)
    if (line.match(/^#{1,6}\s/)) {
      const headingLevel = (line.match(/^#+/)?.[0]?.length || 1);
      const headingText = line.replace(/^#+\s*/, '');
      const html = `<h${Math.min(headingLevel, 6)} class="markdown-heading">${headingText}</h${Math.min(headingLevel, 6)}>`;
      
      decorations.push(
        Decoration.widget({
          widget: new MarkdownPreviewWidget(html),
          side: 1
        }).range(lineStart)
      );
    }
    
    // Preview code blocks
    if (line.trim().startsWith('```')) {
      const codeStart = i;
      while (i < lines.length && !lines[i].startsWith('```')) i++;
      const codeEnd = i;
      const codeContent = lines.slice(codeStart + 1, codeEnd).join('\n');
      const codeBlock = `<pre class="markdown-code-block"><code>${codeContent}</code></pre>`;
      
      decorations.push(
        Decoration.widget({
          widget: new MarkdownPreviewWidget(codeBlock),
          side: 1
        }).range(lineStart)
      );
    }
    
    pos = lineEnd + 1; // +1 for newline
  }
  
  return Decoration.set(decorations);
}
```

### **Step 4.1: Create Markdown Parser Utility (Optional)**
**File**: `/src/utils/markdown-parser.ts`
```typescript
import { marked } from 'marked';
import hljs from 'highlight.js';

// Configure marked for better parsing
marked.setOptions({
  highlight: function(code: string, lang: string) {
    if (lang && hljs.getLanguage(lang)) {
      try {
        return hljs.highlight(code, { language: lang }).value;
      } catch (err) {
        console.warn(`Highlight.js error for language ${lang}:`, err);
      }
    }
    return hljs.highlightAuto(code).value;
  },
  gfm: true,
  breaks: true,
  pedantic: false,
  smartLists: true,
  smartypants: true
});

// Custom renderer for better control
const renderer = new marked.Renderer();

// Override heading renderer
renderer.heading = function(text: string, level: number) {
  const escapedText = text.toLowerCase().replace(/[^\w]+/g, '-');
  return `
    <h${level} id="${escapedText}" class="markdown-heading">
      ${text}
    </h${level}>
  `;
};

// Override code renderer
renderer.code = function(code: string, language: string | undefined, isEscaped: boolean) {
  const validLang = language && hljs.getLanguage(language) ? language : 'plaintext';
  const highlighted = hljs.highlight(code, { language: validLang }).value;
  return `
    <pre class="markdown-code-block language-${validLang}">
      <code class="hljs language-${validLang}">${highlighted}</code>
    </pre>
  `;
};

marked.use({ renderer });

export interface MarkdownBlock {
  from: number;
  to: number;
  html: string;
  type: 'heading' | 'paragraph' | 'code' | 'list' | 'blockquote' | 'hr';
}

export function parseMarkdownForInlinePreview(markdown: string): MarkdownBlock[] {
  const blocks: MarkdownBlock[] = [];
  const lines = markdown.split('\n');
  let currentPos = 0;
  
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];
    const from = currentPos;
    const to = from + line.length;
    
    // Parse line type
    if (line.startsWith('# ')) {
      blocks.push({
        from,
        to,
        html: marked.parse(line) as string,
        type: 'heading'
      });
    } else if (line.startsWith('## ')) {
      blocks.push({
        from,
        to,
        html: marked.parse(line) as string,
        type: 'heading'
      });
    } else if (line.startsWith('### ')) {
      blocks.push({
        from,
        to,
        html: marked.parse(line) as string,
        type: 'heading'
      });
    } else if (line.startsWith('```')) {
      // Handle code blocks (multi-line)
      const codeStart = i;
      while (i < lines.length && !lines[i].startsWith('```')) i++;
      const codeEnd = i;
      const codeContent = lines.slice(codeStart + 1, codeEnd).join('\n');
      const lang = lines[codeStart].replace('```', '').trim();
      
      let codeFrom = 0;
      for (let j = 0; j < codeStart; j++) {
        codeFrom += lines[j].length + 1;
      }
      
      let codeTo = codeFrom;
      for (let j = codeStart; j <= codeEnd; j++) {
        codeTo += lines[j]?.length || 0;
        if (j < codeEnd) codeTo += 1; // Newline char
      }
      
      const highlighted = hljs.highlight(codeContent, { 
        language: lang || 'plaintext' 
      }).value;
      
      blocks.push({
        from: codeFrom,
        to: codeTo,
        html: `<pre class="markdown-code-block"><code>${highlighted}</code></pre>`,
        type: 'code'
      });
    } else if (line.trim().startsWith('- ') || line.trim().startsWith('* ') || /^\d+\./.test(line.trim())) {
      let listFrom = 0;
      for (let j = 0; j < i; j++) {
        listFrom += lines[j].length + 1;
      }
      const listTo = listFrom + line.length;
      
      blocks.push({
        from: listFrom,
        to: listTo,
        html: marked.parse(line) as string,
        type: 'list'
      });
    } else if (line.trim().startsWith('>')) {
      let quoteFrom = 0;
      for (let j = 0; j < i; j++) {
        quoteFrom += lines[j].length + 1;
      }
      const quoteTo = quoteFrom + line.length;
      
      blocks.push({
        from: quoteFrom,
        to: quoteTo,
        html: marked.parse(line) as string,
        type: 'blockquote'
      });
    } else if (line.trim() === '---' || line.trim() === '***' || line.trim() === '___') {
      let hrFrom = 0;
      for (let j = 0; j < i; j++) {
        hrFrom += lines[j].length + 1;
      }
      const hrTo = hrFrom + line.length;
      
      blocks.push({
        from: hrFrom,
        to: hrTo,
        html: '<hr class="markdown-hr" />',
        type: 'hr'
      });
    } else if (line.trim().length > 0) {
      let paraFrom = 0;
      for (let j = 0; j < i; j++) {
        paraFrom += lines[j].length + 1;
      }
      const paraTo = paraFrom + line.length;
      
      blocks.push({
        from: paraFrom,
        to: paraTo,
        html: marked.parse(line) as string,
        type: 'paragraph'
      });
    }
    
    return blocks;
}

export function renderFullMarkdown(markdown: string): string {
  return marked.parse(markdown) as string;
}
```

// Simple toggle between edit and preview modes
class MarkdownEditorWithPreview {
  private editView: EditorView;
  private previewElement: HTMLElement;
  private mode: 'edit' | 'preview' = 'edit';
  private container: HTMLElement;
  
  constructor(host: HTMLElement, content: string, path: string) {
    this.container = host;
    
    // Create editor view
    this.editView = new EditorView({
      state: EditorState.create({
        doc: content,
        extensions: [/* ... */]
      }),
      parent: host
    });
    
    // Create preview element (hidden initially)
    this.previewElement = document.createElement('div');
    this.previewElement.className = 'markdown-full-preview';
    this.previewElement.style.display = 'none';
    host.appendChild(this.previewElement);
    
    // Add toggle button
    const toggleBtn = document.createElement('button');
    toggleBtn.className = 'preview-toggle-btn';
    toggleBtn.textContent = 'Preview';
    toggleBtn.onclick = () => this.toggleMode();
    host.appendChild(toggleBtn);
  }
  
  toggleMode() {
    if (this.mode === 'edit') {
      this.showPreview();
    } else {
      this.showEditor();
    }
  }
  
  private showPreview() {
    const content = this.editView.state.doc.toString();
    this.previewElement.innerHTML = renderFullMarkdown(content);
    this.editView.dom.style.display = 'none';
    this.previewElement.style.display = 'block';
    this.mode = 'preview';
  }
  
  private showEditor() {
    this.editView.dom.style.display = 'block';
    this.previewElement.style.display = 'none';
    this.mode = 'edit';
  }
}
```
